What is Project Gutenberg?

Project Gutenberg is the oldest digital library, founded in 1971 by Michael Hart to digitize and archive cultural works, particularly public-domain books, to encourage eBook creation and distribution. It is a volunteer-driven nonprofit organization offering over 75,000 free eBooks in various formats, available for download or online reading. The collection focuses on older works whose U.S. copyright has expired, and the project’s mission is to make literature freely accessible to everyone.

# 전체 단어 수
total_words <- length(word_list)

# 몇 구간으로 나눌지 설정 (예: 100구간)
num_bins <- 100
words_per_bin <- ceiling(total_words / num_bins)

# 각 단어에 인덱스 부여하고 'lonely' 여부 판단
df <- tibble(
  word = word_list,
  position = 1:total_words,
  bin = ceiling(position / words_per_bin)
) %>%
  filter(word == "lonely") %>%
  count(bin)

# 시각화

quotes <- JaneEyre0 %>%
  filter(str_detect(text, '^\".*\"$'))  # 큰따옴표로 시작하고 끝나는 줄만 추출

# 2. 대사를 단어 단위로 토큰화
quote_words <- quotes %>%
  unnest_tokens(word, text) %>%       # 단어로 나누기
  filter(str_detect(word, "^[a-z']+$")) %>%  # 영어 단어만 (숫자/기호 제외)
  filter(!word %in% stop_words$word)        # 불용어 제거

# 3. 단어 빈도 계산
top_words <- quote_words %>%
  count(word, sort = TRUE) %>%
  slice_max(n, n = 15)   # 상위 15개 단어 선택

# 4. 시각화: 막대 그래프
